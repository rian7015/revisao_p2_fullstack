Setup Básico

HTML:
<canvas id="canvas" width="400" height="400"></canvas>

CSS:
#canvas { background-color: #eee; }

JavaScript:
let canvas = document.getElementById('canvas');
let ctx = canvas.getContext('2d');

Coordenadas: (0,0) = canto superior esquerdo
X cresce →  Y cresce ↓

Formas Geométricas

Retângulos

ctx.beginPath();
ctx.fillStyle = 'blue';
ctx.strokeStyle = 'red';
ctx.lineWidth = 3;
ctx.fillRect(x, y, w, h);   // preenchido
ctx.strokeRect(x, y, w, h); // borda
ctx.clearRect(x, y, w, h);  // apaga área
ctx.closePath();

Círculos / Arcos

ctx.beginPath();
ctx.arc(cx, cy, raio,
        anguloInicial, anguloFinal);
// Círculo completo:
ctx.arc(100, 75, 50, 0, 2*Math.PI);
ctx.fill();
ctx.stroke();
ctx.closePath();

Linhas

ctx.beginPath();
ctx.moveTo(x1, y1); // ponto inicial
ctx.lineTo(x2, y2); // ponto final
ctx.lineTo(x3, y3); // continua
ctx.stroke();       // desenha
ctx.fill();         // preenche
ctx.closePath();

Texto

ctx.font = "30px Arial";
ctx.textAlign = "center";
ctx.fillStyle = "blue";
ctx.fillText("Texto", x, y);
ctx.strokeText("Texto", x, y);

Objetos no Canvas

let retangulo = {
  x: 0, y: 0,
  largura: 50, altura: 50,
  cor: "red",
  desenha: function() {
    ctx.beginPath();
    ctx.fillStyle = this.cor;
    ctx.fillRect(this.x, this.y, this.largura, this.altura);
    ctx.closePath();
  }
};

let circulo = {
  x: 100, y: 200,
  raio: 30,
  cor: "blue",
  desenha: function() {
    ctx.beginPath();
    ctx.fillStyle = this.cor;
    ctx.arc(this.x, this.y, this.raio, 0, 2 * Math.PI);
    ctx.fill();
    ctx.closePath();
  }
};

Animação com teclado — Use as setas ← ↑ → ↓

function animacao() {
  ctx.clearRect(0, 0, canvas.width, canvas.height); // LIMPA
  objeto.x += velocidade;                            // MOVE
  objeto.desenha();                                  // DESENHA
  requestAnimationFrame(animacao);                   // REPETE (~60fps)
}
animacao(); // inicia

document.addEventListener('keydown', function(evento) {
  let tecla = evento.key;
  if (tecla == 'ArrowUp')    objeto.y -= 5;
  if (tecla == 'ArrowDown')  objeto.y += 5;
  if (tecla == 'ArrowLeft')  objeto.x -= 5;
  if (tecla == 'ArrowRight') objeto.x += 5;
});

Seguindo o Mouse

document.addEventListener('mousemove', function(evento) {
  let rect = canvas.getBoundingClientRect();
  let mx = evento.clientX - rect.left;  // posição X dentro do canvas
  let my = evento.clientY - rect.top;   // posição Y dentro do canvas
  objeto.x = mx;
  objeto.y = my;
});

Funções úteis de Animação

requestAnimationFrame(funcao)  // melhor! ~60fps, para quando aba fica inativa
setInterval(funcao, 16)        // executa a cada 16ms (~60fps)
setTimeout(funcao, 1000)       // executa UMA vez após 1 segundo

ctx.clearRect(0, 0, canvas.width, canvas.height); // limpa canvas inteiro
