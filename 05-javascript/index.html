1. Variáveis e Tipos

let nome = "João";         // string
let idade = 20;            // number (inteiro)
let altura = 1.75;         // number (decimal)
let ativo = true;          // boolean
const PI = 3.14159;        // constante (não muda)

// Escopos:
x = 10;          // global - NÃO RECOMENDADO
var a = 10;      // escopo de função
let b = 10;      // escopo de bloco ✅
const c = 10;    // bloco, somente leitura ✅

2. Operadores

// Aritméticos
+  -  *  /  %   (módulo/resto)

// Comparação (use sempre ===)
===  !==   >   <   >=   <=
==   !=    (converte tipo - evite!)

// Lógicos
&&  (E)    ||  (OU)    !  (NÃO)

// Exemplos:
5 + 3      // 8
10 % 3     // 1
"5" === 5  // false (tipos diferentes)
"5" == 5   // true  (converte tipo)

3. Conversão de Tipos

let num = Number("42");        // "42" → 42
let inteiro = parseInt("42.9"); // → 42
let decimal = parseFloat("3.14"); // → 3.14
let texto = String(42);        // 42 → "42"
let texto2 = (42).toString();  // 42 → "42"

4. Controle de Fluxo

// IF / ELSE IF / ELSE
if (nota >= 7) {
  console.log("Aprovado");
} else if (nota >= 5) {
  console.log("Recuperação");
} else {
  console.log("Reprovado");
}

// WHILE
let i = 0;
while (i < 5) {
  console.log(i);
  i++;
}

// FOR
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// FOR com array
let frutas = ["maçã", "banana", "uva"];
for (let i = 0; i < frutas.length; i++) {
  console.log(frutas[i]);
}

5. Funções

// Sem parâmetro, sem retorno
function ola() {
  alert("Olá!");
}

// Com parâmetro
function saudacao(nome) {
  alert("Olá, " + nome + "!");
}

// Com parâmetros e retorno
function soma(a, b) {
  return a + b;
}
let resultado = soma(5, 3); // 8

// Chamadas
ola();
saudacao("Maria");
console.log(soma(10, 20)); // 30

6. Arrays

let frutas = ["maçã", "banana", "uva"];
frutas.push("laranja");   // adiciona ao final
frutas.pop();             // remove o último
frutas.length;            // tamanho: 3
frutas[0];                // "maçã" (índice 0)
delete frutas[1];         // remove índice 1

7. DOM — Manipulando o HTML

// Pegar elemento por ID
let el = document.getElementById("meuId");

// Alterar conteúdo
el.innerHTML = "novo conteúdo";

// Alterar estilo
el.style.color = "red";
el.style.backgroundColor = "yellow";
el.style.display = "none";     // esconde
el.style.display = "block";    // mostra
